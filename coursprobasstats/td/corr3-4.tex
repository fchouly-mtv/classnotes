\begin{corrige}{3-4}

\begin{enumerate}
 \item On remarque que:
 \begin{eqnarray*}
  P ( A \cap B )& = & P(A) + P(B) - P(A\cup B)\\
  & \geq & 0.53 + 0.72 - 1 \geq 0.25 > 0.
 \end{eqnarray*}
 Donc $A$ et $B$ ne peuvent pas être incompatibles.
 
 \item On a tout d'abord:
 \[
  P ( A \cap B ) = P(A) + P(B) - P(A\cup B) = 0.36.
 \]
 Et puis:
 \[
  P ( A \cap \bar{B} ) = P ( A ) - P ( A \cap B) = 0.17.
 \]

\end{enumerate}


\end{corrige}
