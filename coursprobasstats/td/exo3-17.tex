\begin{exercice}\label{exo3-17} (\boringexo)

\begin{enumerate}
 \item On tire deux cartes l'une apr\`es l'autre, sans remise, dans un jeu de 32 cartes.\\
Calculer la probabilit\'e de :
\begin{enumerate}
 \item tirer un as, puis un roi,
 \item tirer une dame, puis un coeur. %\\
\end{enumerate}

\item Refaire le calcul dans le cas d'un tirage avec remise. %\\

\item On tire maintenant deux cartes simultan\'ement dans un jeu de 32 cartes.\\
Calculer la probabilit\'e de :

\begin{enumerate}
 \item tirer un as et un roi,
 \item tirer une dame et un coeur. %\\
\end{enumerate}

\end{enumerate}

\end{exercice}
