\begin{exercice}\label{exo4-12} (\minsyndical)

\begin{enumerate}

\item Soit $X$ une variable qui suit la loi normale $\mathcal{N}(5,2)$. Calculer:
\[
P(4 \leq X \leq 6) ; \quad P(X \geq 10) ; \quad P ( 5.8 \leq X \leq 6.2).
\]

\item Soit $X$ une variable qui suit la loi normale $\mathcal{N}(-10,0.1)$. Calculer:
\[
P(-11.5 \leq X \leq -8.5) ; \quad P(X \leq -8) ; \quad P ( -7.8 \leq X \leq -7.6).
\]


\end{enumerate}
