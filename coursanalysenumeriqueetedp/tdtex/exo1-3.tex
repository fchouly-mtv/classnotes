\begin{exercice}\label{exo1-3}

\begin{enumerate}

\item Montrer que pour toute fonction $w \in \mathcal{C}^0 ( [0,1] )$, telle que :

\[
\int_0^1 w(x) v(x) dx = 0, \quad \forall v \in V,
\]

alors $w(x)=0$ pour $0 \leq x \leq 1$.

\item Soit $u$ la solution du problème faible associé à \eqref{eq_model}. On fait l'hypothèse de régularité suivante : $\frac{d^2 u}{dx^2}$ existe et est continue sur $[0,1]$. Alors montrer que $u$
est aussi solution du problème fort \eqref{eq_model}.\\

\end{enumerate}
